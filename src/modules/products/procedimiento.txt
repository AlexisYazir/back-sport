PROCESO PARA CALCULAR EL PRECIO ACTUAL DE UN PRODUCTO

7. Ejemplo completo

Inventario actual:

stock = 80
costo_promedio = 90

Compras:

20 unidades a 120

Cálculo:

(80*90 + 20*120) / 100
(7200 + 2400) / 100
9600 / 100

nuevo_promedio = 96

Resultado:

stock = 100
costo_promedio = 96

Flujo correcto en tu sistema
1️⃣ Crear producto base

Primero registras el producto general.

Tabla:

products

Ejemplo:

Tenis Nike Air Max

Aquí solo va información general:

nombre

descripción

marca

categoría

2️⃣ Crear variantes del producto

Después creas las variantes que realmente se venden.

Tabla:

product_variants

Ejemplo:

variante	SKU	precio
Rojo 42	NK-AIR-R42	1800
Rojo 43	NK-AIR-R43	1800

Aquí defines:

SKU

precio de venta

atributos

Todavía no hay inventario.

3️⃣ Crear registro inicial en inventario

Cuando creas una variante, normalmente también creas su fila en:

inventory

con stock inicial 0.

Ejemplo:

variante	stock	costo_promedio
NK-AIR-R42	0	0

Esto prepara la variante para recibir movimientos.

4️⃣ Registrar compra a proveedor

Ahora sí entra el inventario real.

Aquí registras una entrada de inventario en:

inventory_movements

Ejemplo:

tipo: entrada
variante: NK-AIR-R42
cantidad: 20
costo_unitario: 900
referencia: compra_proveedor
5️⃣ El movimiento actualiza inventario

Tu backend debe hacer:

inventory.stock_actual += cantidad

y recalcular:

costo_promedio

Resultado:

variante	stock	costo_promedio
NK-AIR-R42	20	900

create table roles(id_rol integer generated always as identity primary key, rol varchar(20));

-- Relacion de roles con usuarios
alter table users add constraint fk_users_roles foreign key (rol) references roles(id_rol);

-- insert de los roles de usuario
insert into roles(rol) values ('Usuario'),('Empleado'),('Administrador');

-- 3. Tabla categorias
CREATE TABLE categories (
    id_categoria INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE,
    id_padre INTEGER REFERENCES categories(id_categoria) -- Para subcategorías (ej. Ropa -> Playeras)
);

-- 4. Tabla marcas
CREATE TABLE marcas (
    id_marca INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE,
    sitio_web VARCHAR(255)
);

-- 5. Tabla Producto "Base"
CREATE TABLE products (
    id_producto INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_marca INTEGER REFERENCES marcas(id_marca) ON DELETE CASCADE,
    id_categoria INTEGER REFERENCES categories(id_categoria) ON DELETE SET NULL,
    nombre VARCHAR(150) NOT NULL,
    descripcion TEXT,
    activo BOOLEAN DEFAULT TRUE,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 6. Tabla Definición de Atributos (Ej: 'Talla', 'Color', 'Material', 'Peso')
CREATE TABLE attributes (
    id_atributo INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE
);

-- 7. Variantes del Producto
-- Aquí es donde vive el precio y el stock.
CREATE TABLE product_variants (
    id_variante INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_producto INTEGER REFERENCES products(id_producto) ON DELETE CASCADE,
    sku VARCHAR(50) UNIQUE NOT NULL, -- Código único de inventario
    precio NUMERIC(12, 2) NOT NULL DEFAULT 0,
    stock INTEGER NOT NULL DEFAULT 0,
    imagenes JSONB DEFAULT '[]'::jsonb
);

-- 8. Tabla variante atributos values, Relación Variante - Atributo (Aquí ocurre la magia)
-- Vincula la variante con sus características específicas
CREATE TABLE variant_attribute_values (
    id_variante INTEGER REFERENCES product_variants(id_variante) ON DELETE CASCADE,
    id_atributo INTEGER REFERENCES attributes(id_atributo),
    valor VARCHAR(100) NOT NULL, -- Ej: 'Grande', 'Rojo', 'Fibra de carbono'
    PRIMARY KEY (id_variante, id_atributo)
);

-- 9. Tabla de direcciones
CREATE TABLE direcciones (
    id_direccion INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_usuario INTEGER REFERENCES users(id_usuario) ON DELETE CASCADE,
    alias VARCHAR(50), -- Casa, Trabajo, etc
    calle VARCHAR(150) NOT NULL,
    numero VARCHAR(20),
    colonia VARCHAR(100),
    ciudad VARCHAR(100) NOT NULL,
    estado VARCHAR(100) NOT NULL,
    codigo_postal VARCHAR(10) NOT NULL,
    pais VARCHAR(100) DEFAULT 'México',
    principal BOOLEAN DEFAULT FALSE,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

--10. Tabla de carrito de compras - para que sea persistente
CREATE TABLE carts (
    id_carrito INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_usuario INTEGER REFERENCES users(id_usuario) ON DELETE CASCADE,
    estado VARCHAR(20) DEFAULT 'activo', -- activo, convertido, abandonado
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 11. Tabla de items de carrito
CREATE TABLE cart_items (
    id_carrito INTEGER REFERENCES carts(id_carrito) ON DELETE CASCADE,
    id_variante INTEGER REFERENCES product_variants(id_variante),
    cantidad INTEGER NOT NULL CHECK (cantidad > 0),
    precio_unitario NUMERIC(12,2) NOT NULL,
    PRIMARY KEY (id_carrito, id_variante)
);

-- 12. Tabla de ordenes
CREATE TABLE orders (
    id_orden INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_usuario INTEGER REFERENCES users(id_usuario),
    id_direccion_envio INTEGER REFERENCES direcciones(id_direccion),
    estado VARCHAR(30) DEFAULT 'pendiente', -- pendiente, pagado, enviado, entregado, cancelado

    subtotal NUMERIC(12,2) NOT NULL,
    descuento NUMERIC(12,2) DEFAULT 0,
    total NUMERIC(12,2) NOT NULL,

    metodo_pago VARCHAR(50),
    fecha_pago TIMESTAMP,
    fecha_envio TIMESTAMP,
    fecha_entrega TIMESTAMP,

    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 13. Items de la orden, Esto es el historial de compra real, aquí se congela todo. 
-- Porque si el producto cambia en el futuro, la orden NO debe cambiar.
CREATE TABLE order_items (
    id_orden INTEGER REFERENCES orders(id_orden) ON DELETE CASCADE,
    id_variante INTEGER REFERENCES product_variants(id_variante),
    nombre_producto VARCHAR(150) NOT NULL,
    sku VARCHAR(50),
    cantidad INTEGER NOT NULL,
    precio_unitario NUMERIC(12,2) NOT NULL,
    total NUMERIC(12,2) NOT NULL,
    PRIMARY KEY (id_orden, id_variante)
);

-- 14. Tabla de pagos 
CREATE TABLE pagos (
    id_pago INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_orden INTEGER REFERENCES orders(id_orden) ON DELETE CASCADE,
    proveedor_pago VARCHAR(50), -- Stripe, MercadoPago, PayPal
    referencia_externa VARCHAR(255),
    monto NUMERIC(12,2),
    estado VARCHAR(30), -- aprobado, rechazado, pendiente
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 15. Reseñas para un producto
CREATE TABLE reviews (
    id_review INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_producto INTEGER REFERENCES products(id_producto) ON DELETE CASCADE,
    id_usuario INTEGER REFERENCES users(id_usuario),
    calificacion INTEGER CHECK (calificacion BETWEEN 1 AND 5),
    comentario TEXT,
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 16. Movimientos de inventario, esto para poder llevar un control de lo que son las ventas o movimientos de productos  
CREATE TABLE inventory_movements (
    id_movimiento INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_variante INTEGER REFERENCES product_variants(id_variante),
    tipo VARCHAR(20), -- entrada, salida, ajuste
    cantidad INTEGER NOT NULL,
    referencia VARCHAR(100), -- id_orden, compra proveedor, ajuste manual
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Indices, para rendimiento
CREATE INDEX idx_products_categoria ON products(id_categoria);
CREATE INDEX idx_products_marca ON products(id_marca);
CREATE INDEX idx_variants_producto ON product_variants(id_producto);
CREATE INDEX idx_orders_usuario ON orders(id_usuario);
CREATE INDEX idx_order_estado ON orders(estado);

-- Funcion 0
CREATE OR REPLACE FUNCTION get_all_products()
RETURNS TABLE (
    id_producto INTEGER,
    producto VARCHAR,
    descripcion TEXT,
    activo BOOLEAN,
    fecha_creacion TIMESTAMP,
    marca VARCHAR,
	imagen_marca VARCHAR,
    categoria VARCHAR,
    categoria_padre VARCHAR,
    variantes JSONB
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT 
        p.id_producto,
        p.nombre AS producto,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre AS marca,
		m.imagen,
        c.nombre AS categoria,
        cp.nombre AS categoria_padre,

        COALESCE(
            jsonb_agg(
                DISTINCT jsonb_build_object(
                    'id_variante', v.id_variante,
                    'sku', v.sku,
                    'precio', v.precio,
                    'stock', v.stock,
                    'imagenes', v.imagenes,
                    'atributos',
                        (
                            SELECT jsonb_object_agg(a.nombre, vav.valor)
                            FROM variant_attribute_values vav
                            JOIN attributes a 
                                ON a.id_atributo = vav.id_atributo
                            WHERE vav.id_variante = v.id_variante
                        )
                )
            ) FILTER (WHERE v.id_variante IS NOT NULL),
            '[]'::jsonb
        ) AS variantes

    FROM products p
    LEFT JOIN marcas m 
        ON m.id_marca = p.id_marca
    LEFT JOIN categories c 
        ON c.id_categoria = p.id_categoria
    LEFT JOIN categories cp 
        ON cp.id_categoria = c.id_padre
    LEFT JOIN product_variants v 
        ON v.id_producto = p.id_producto

    GROUP BY 
        p.id_producto,
        p.nombre,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre,
		m.imagen,
        c.nombre,
        cp.nombre;

END;
$$;

-- Funcion 1
CREATE OR REPLACE FUNCTION get_inventory_products()
RETURNS TABLE (
    id_producto        INT,
    producto           TEXT,
    activo             BOOLEAN,
    precio             NUMERIC,
    stock              NUMERIC,
    marca              TEXT,
	imagen			   TEXT,
    fecha_creacion     TIMESTAMP
)
LANGUAGE sql
AS $$
    SELECT 
        t.id_producto,
        t.producto,
        t.activo,
        t.precio,
        t.stock,
        t.marca,
		t.imagen,
        t.fecha_creacion
    FROM (
        SELECT DISTINCT ON (p.id_producto)
            p.id_producto,
            p.nombre AS producto,
            p.activo,
            v.precio,
            v.stock,
            m.nombre AS marca,
			m.imagen,
            p.fecha_creacion,

            -- prioridad interna
            CASE
                WHEN p.activo = false THEN 1
                WHEN v.stock < 5 THEN 2
                ELSE 3
            END AS prioridad

        FROM products p
        LEFT JOIN marcas m 
            ON m.id_marca = p.id_marca
        LEFT JOIN product_variants v 
            ON v.id_producto = p.id_producto

        -- elegimos variante con menor stock
        ORDER BY p.id_producto, v.stock ASC
    ) t
    ORDER BY t.prioridad, t.stock ASC;
$$;

--Consulta los procuctos con variantes, pero sin atributos
CREATE OR REPLACE FUNCTION get_products_with_variants_without_attributes()
RETURNS TABLE (
    id_producto INT,
    nombre TEXT,
    descripcion TEXT,
    activo BOOLEAN,
    fecha_creacion TIMESTAMP
)
LANGUAGE sql
AS $$
    SELECT 
        p.id_producto,
        p.nombre,
        p.descripcion,
        p.activo,
        p.fecha_creacion
    FROM products p
    WHERE EXISTS (
        SELECT 1
        FROM product_variants v
        WHERE v.id_producto = p.id_producto
    )
    AND EXISTS (
        SELECT 1
        FROM product_variants v
        WHERE v.id_producto = p.id_producto
          AND NOT EXISTS (
              SELECT 1
              FROM variant_attribute_values vav
              WHERE vav.id_variante = v.id_variante
          )
    ) order by id_producto;
$$;

-- Funcion 5
-- funcion para actualizar datos de inventario
CREATE OR REPLACE FUNCTION update_product_variant(
    p_id_producto INT,
    p_id_variante INT,
    p_precio NUMERIC,
    p_stock INT,
    p_activo BOOLEAN
)
RETURNS INT
LANGUAGE plpgsql
AS $$
DECLARE
    v_rows_affected INT := 0;
    v_temp INT;
BEGIN

    -- Update variante
    UPDATE product_variants
    SET 
        precio = p_precio,
        stock = p_stock
    WHERE id_variante = p_id_variante
      AND id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    -- Update producto
    UPDATE products
    SET 
        activo = p_activo
    WHERE id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    RETURN v_rows_affected;

END;
$$;

-- Funcion 7
-- Actualiza los datos de una variante
CREATE OR REPLACE FUNCTION update_product_var_attr(
    p_id_producto INT,

    p_id_variante INT,
    p_sku TEXT,
    p_imagenes JSONB,

    p_id_atributo INT,
    p_valor TEXT
)
RETURNS TABLE (
    updated_variant INT,
    updated_attribute INT
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_rows_variant INT := 0;
    v_rows_attribute INT := 0;
BEGIN

    -- Update variante
    UPDATE product_variants
    SET sku = p_sku,
        imagenes = p_imagenes
    WHERE id_variante = p_id_variante
      AND id_producto = p_id_producto;

    GET DIAGNOSTICS v_rows_variant = ROW_COUNT;

    -- Update atributo
    UPDATE variant_attribute_values
    SET valor = p_valor
    WHERE id_variante = p_id_variante
      AND id_atributo = p_id_atributo;

    GET DIAGNOSTICS v_rows_attribute = ROW_COUNT;

    RETURN QUERY
    SELECT v_rows_variant, v_rows_attribute;
END;
$$;

        m.imagen,

        cp.nombre AS categoria_padre,
        COALESCE(
            jsonb_agg(
                DISTINCT jsonb_build_object(
                    'id_variante', v.id_variante,
                    'sku', v.sku,
                    'precio', v.precio,
                    'stock', v.stock,
                    'imagenes', v.imagenes,
                    'atributos',
                        (
                            SELECT jsonb_object_agg(a.nombre, vav.valor)
                            FROM core.variant_attribute_values vav
                            JOIN core.attributes a 
                                ON a.id_atributo = vav.id_atributo
                            WHERE vav.id_variante = v.id_variante
                        )
                )
            ) FILTER (WHERE v.id_variante IS NOT NULL),
            '[]'::jsonb
        )
    FROM core.products p
    LEFT JOIN core.marcas m 
        ON m.id_marca = p.id_marca
    LEFT JOIN core.categories c 
        ON c.id_categoria = p.id_categoria
    LEFT JOIN core.categories cp 
        ON cp.id_categoria = c.id_padre
    LEFT JOIN core.product_variants v 
        ON v.id_producto = p.id_producto
    GROUP BY 
        p.id_producto,
        p.nombre,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre,
        m.imagen,
        c.nombre,
        cp.nombre;
END;
$$;

-- ===============================
CREATE OR REPLACE FUNCTION core.get_inventory_products()
RETURNS TABLE (
    id_producto INT,
    producto TEXT,
    activo BOOLEAN,
    precio NUMERIC,
    stock NUMERIC,
    marca TEXT,
    imagen TEXT,
    fecha_creacion TIMESTAMP
)
LANGUAGE sql
AS $$
    SELECT 
        t.id_producto,
        t.producto,
        t.activo,
        t.precio,
        t.stock,
        t.marca,
        t.imagen,
        t.fecha_creacion
    FROM (
        SELECT DISTINCT ON (p.id_producto)
            p.id_producto,
            p.nombre AS producto,
            p.activo,
            v.precio,
            v.stock,
            m.nombre AS marca,
            m.imagen,
            p.fecha_creacion,
            CASE
                WHEN p.activo = false THEN 1
                WHEN v.stock < 5 THEN 2
                ELSE 3
            END AS prioridad
        FROM core.products p
        LEFT JOIN core.marcas m 
            ON m.id_marca = p.id_marca
        LEFT JOIN core.product_variants v 
            ON v.id_producto = p.id_producto
        ORDER BY p.id_producto, v.stock ASC
    ) t
    ORDER BY t.prioridad, t.stock ASC;
$$;

-- ===============================
CREATE OR REPLACE FUNCTION core.update_product_variant(
    p_id_producto INT,
    p_id_variante INT,
    p_precio NUMERIC,
    p_stock INT,
    p_activo BOOLEAN
)
RETURNS INT
LANGUAGE plpgsql
AS $$
DECLARE
    v_rows_affected INT := 0;
    v_temp INT;
BEGIN
    UPDATE core.product_variants
    SET precio = p_precio,
        stock = p_stock
    WHERE id_variante = p_id_variante
      AND id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    UPDATE core.products
    SET activo = p_activo
    WHERE id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    RETURN v_rows_affected;
END;
$$;

    id_variante INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,

    stock INTEGER NOT NULL DEFAULT 0,

    id_variante INTEGER REFERENCES product_variants(id_variante) ON DELETE CASCADE,
    id_atributo INTEGER REFERENCES attributes(id_atributo),

CREATE OR REPLACE FUNCTION get_all_products()
RETURNS TABLE (
    id_producto INTEGER,
    producto VARCHAR,
    descripcion TEXT,
    activo BOOLEAN,
    fecha_creacion TIMESTAMP,
    marca VARCHAR,
	imagen_marca VARCHAR,
    categoria VARCHAR,
    categoria_padre VARCHAR,
    variantes JSONB
)

AS $$
BEGIN
    RETURN QUERY
    SELECT 
        p.id_producto,
        p.nombre AS producto,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre AS marca,
		m.imagen,
        c.nombre AS categoria,
        cp.nombre AS categoria_padre,

        COALESCE(
            jsonb_agg(
                DISTINCT jsonb_build_object(
                    'id_variante', v.id_variante,
                    'sku', v.sku,
                    'precio', v.precio,
                    'stock', v.stock,
                    'imagenes', v.imagenes,
                    'atributos',
                        (
                            SELECT jsonb_object_agg(a.nombre, vav.valor)
                            FROM variant_attribute_values vav
                            JOIN attributes a 
                                ON a.id_atributo = vav.id_atributo
                            WHERE vav.id_variante = v.id_variante
                        )
                )
            ) FILTER (WHERE v.id_variante IS NOT NULL),

        p.activo,
        p.fecha_creacion,
        m.nombre,
		m.imagen,

CREATE OR REPLACE FUNCTION get_inventory_products()
RETURNS TABLE (
    id_producto        INT,
    producto           TEXT,
    activo             BOOLEAN,
    precio             NUMERIC,
    stock              NUMERIC,
    marca              TEXT,
	imagen			   TEXT,
    fecha_creacion     TIMESTAMP
)
LANGUAGE sql
AS $$
    SELECT 
        t.id_producto,
        t.producto,
        t.activo,
        t.precio,
        t.stock,
        t.marca,
		t.imagen,
        t.fecha_creacion
    FROM (
        SELECT DISTINCT ON (p.id_producto)
            p.id_producto,
            p.nombre AS producto,
            p.activo,
            v.precio,

            m.nombre AS marca,
			m.imagen,

            -- prioridad interna
            CASE
                WHEN p.activo = false THEN 1
                WHEN v.stock < 5 THEN 2
                ELSE 3

        LEFT JOIN marcas m 
            ON m.id_marca = p.id_marca
        LEFT JOIN product_variants v 
            ON v.id_producto = p.id_producto

        -- elegimos variante con menor stock
        ORDER BY p.id_producto, v.stock ASC
    ) t
    ORDER BY t.prioridad, t.stock ASC;

CREATE OR REPLACE FUNCTION core.get_all_products()
RETURNS TABLE (
    id_producto INTEGER,
    producto VARCHAR,
    descripcion TEXT,
    activo BOOLEAN,
    fecha_creacion TIMESTAMP,
    marca VARCHAR,
    imagen_marca VARCHAR,
    categoria VARCHAR,
    categoria_padre VARCHAR,
    variantes JSONB
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT 
        p.id_producto,
        p.nombre AS producto,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre AS marca,
        m.imagen,
        c.nombre AS categoria,
        cp.nombre AS categoria_padre,
        COALESCE(
            jsonb_agg(
                DISTINCT jsonb_build_object(
                    'id_variante', v.id_variante,
                    'sku', v.sku,
                    'precio', v.precio,
                    'stock', vi.stock_actual,
                    'imagenes', v.imagenes,
                    'atributos',
                        (
                            SELECT jsonb_object_agg(a.nombre, vav.valor)
                            FROM core.variant_attribute_values vav
                            JOIN core.attributes a 
                                ON a.id_atributo = vav.id_atributo
                            WHERE vav.id_variante = v.id_variante
                        )
                )
            ) FILTER (WHERE v.id_variante IS NOT NULL),
            '[]'::jsonb
        )
    FROM core.products p
    LEFT JOIN core.marcas m 
        ON m.id_marca = p.id_marca
    LEFT JOIN core.categories c 
        ON c.id_categoria = p.id_categoria
    LEFT JOIN core.categories cp 
        ON cp.id_categoria = c.id_padre
    LEFT JOIN core.product_variants v 
        ON v.id_producto = p.id_producto
		AND v.precio >0
	INNER JOIN core.inventory vi
    	ON vi.id_variante = v.id_variante
   		AND vi.stock_actual > 0
    GROUP BY 
        p.id_producto,
        p.nombre,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre,
        m.imagen,
        c.nombre,
        cp.nombre;
END;
$$;

CREATE OR REPLACE FUNCTION core.get_inventory_products()
RETURNS TABLE (
    id_producto INT,
    producto TEXT,
    activo BOOLEAN,
    precio NUMERIC,
    stock NUMERIC,
    marca TEXT,
    imagen TEXT,
    fecha_creacion TIMESTAMP
)
LANGUAGE sql
AS $$
    SELECT 
        t.id_producto,
        t.producto,
        t.activo,
        t.precio,
        t.stock_actual,
        t.marca,
        t.imagen,
        t.fecha_creacion
    FROM (
        SELECT DISTINCT ON (p.id_producto)
            p.id_producto,
            p.nombre AS producto,
            p.activo,
            v.precio,
            vi.stock_actual,
            m.nombre AS marca,
            m.imagen,
            p.fecha_creacion,
            CASE
                WHEN p.activo = false THEN 1
                WHEN vi.stock_actual < 5 THEN 2
                ELSE 3
            END AS prioridad
        FROM core.products p
        LEFT JOIN core.marcas m 
            ON m.id_marca = p.id_marca
        LEFT JOIN core.product_variants v 
            ON v.id_producto = p.id_producto
		INNER JOIN core.inventory vi
			ON vi.id_variante = v.id_variante
        ORDER BY p.id_producto, vi.stock_actual ASC
    ) t
    ORDER BY t.prioridad, t.stock_actual ASC;
$$;

CREATE OR REPLACE FUNCTION core.create_inventory_movement(
    p_id_variante INT,
    p_tipo TEXT,
    p_cantidad INT,
    p_costo_unitario NUMERIC(10,2),
    p_referencia_tipo TEXT,
    p_referencia_id INT
)
RETURNS TABLE (
    new_stock INT,
    new_costo_promedio NUMERIC(10,2)
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_stock_actual INT;
    v_costo_promedio NUMERIC(10,2);
    v_new_stock INT;
    v_new_costo NUMERIC(10,2);
BEGIN

    -- Obtener inventario actual
    SELECT stock_actual, costo_promedio
    INTO v_stock_actual, v_costo_promedio
    FROM core.inventory
    WHERE id_variante = p_id_variante;

    -- Si no existe inventario se crea
    IF NOT FOUND THEN
        INSERT INTO core.inventory(id_variante, stock_actual, costo_promedio)
        VALUES(p_id_variante, 0, 0);

        v_stock_actual := 0;
        v_costo_promedio := 0;
    END IF;

    -- Calcular nuevo stock
    IF p_tipo = 'entrada' THEN
        v_new_stock := v_stock_actual + p_cantidad;

        -- nuevo costo promedio ponderado
        v_new_costo := (
            (v_stock_actual * v_costo_promedio) +
            (p_cantidad * p_costo_unitario)
        ) / v_new_stock;

    ELSIF p_tipo = 'salida' THEN
        v_new_stock := v_stock_actual - p_cantidad;

        IF v_new_stock < 0 THEN
            RAISE EXCEPTION 'Stock insuficiente para la variante %', p_id_variante;
        END IF;

        v_new_costo := v_costo_promedio;

    ELSIF p_tipo = 'ajuste' THEN
        v_new_stock := p_cantidad;
        v_new_costo := p_costo_unitario;

    ELSE
        RAISE EXCEPTION 'Tipo de movimiento inválido';
    END IF;

    -- Registrar movimiento
    INSERT INTO core.inventory_movements(
        id_variante,
        tipo,
        cantidad,
        costo_unitario,
        referencia_tipo,
        referencia_id
    )
    VALUES(
        p_id_variante,
        p_tipo,
        p_cantidad,
        p_costo_unitario,
        p_referencia_tipo,
        p_referencia_id
    );

    -- Actualizar inventario
    UPDATE core.inventory
    SET stock_actual = v_new_stock,
        costo_promedio = v_new_costo
    WHERE id_variante = p_id_variante;

    RETURN QUERY
    SELECT v_new_stock, v_new_costo;

END;
$$;

-- 6. Tabla Definición de Atributos (Ej: 'Talla', 'Color', 'Material', 'Peso')
CREATE TABLE attributes (
    id_atributo INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE
);

-- 7. Variantes del Producto
-- Aquí es donde vive el precio y el stock.
CREATE TABLE product_variants (
    id_variante INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_producto INTEGER REFERENCES products(id_producto) ON DELETE CASCADE,
    sku VARCHAR(50) UNIQUE NOT NULL, -- Código único de inventario
    precio NUMERIC(12, 2) NOT NULL DEFAULT 0,
    stock INTEGER NOT NULL DEFAULT 0,
    imagenes JSONB DEFAULT '[]'::jsonb
);

-- 8. Tabla variante atributos values, Relación Variante - Atributo (Aquí ocurre la magia)
-- Vincula la variante con sus características específicas
CREATE TABLE variant_attribute_values (
    id_variante INTEGER REFERENCES product_variants(id_variante) ON DELETE CASCADE,
    id_atributo INTEGER REFERENCES attributes(id_atributo),
    valor VARCHAR(100) NOT NULL, -- Ej: 'Grande', 'Rojo', 'Fibra de carbono'
    PRIMARY KEY (id_variante, id_atributo)
);

-- 9. Tabla de direcciones
CREATE TABLE direcciones (
    id_direccion INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_usuario INTEGER REFERENCES users(id_usuario) ON DELETE CASCADE,
    alias VARCHAR(50), -- Casa, Trabajo, etc
    calle VARCHAR(150) NOT NULL,
    numero VARCHAR(20),
    colonia VARCHAR(100),
    ciudad VARCHAR(100) NOT NULL,
    estado VARCHAR(100) NOT NULL,
    codigo_postal VARCHAR(10) NOT NULL,
    pais VARCHAR(100) DEFAULT 'México',
    principal BOOLEAN DEFAULT FALSE,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

--10. Tabla de carrito de compras - para que sea persistente
CREATE TABLE carts (
    id_carrito INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_usuario INTEGER REFERENCES users(id_usuario) ON DELETE CASCADE,
    estado VARCHAR(20) DEFAULT 'activo', -- activo, convertido, abandonado
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 11. Tabla de items de carrito
CREATE TABLE cart_items (
    id_carrito INTEGER REFERENCES carts(id_carrito) ON DELETE CASCADE,
    id_variante INTEGER REFERENCES product_variants(id_variante),
    cantidad INTEGER NOT NULL CHECK (cantidad > 0),
    precio_unitario NUMERIC(12,2) NOT NULL,
    PRIMARY KEY (id_carrito, id_variante)
);

-- 12. Tabla de ordenes
CREATE TABLE orders (
    id_orden INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_usuario INTEGER REFERENCES users(id_usuario),
    id_direccion_envio INTEGER REFERENCES direcciones(id_direccion),
    estado VARCHAR(30) DEFAULT 'pendiente', -- pendiente, pagado, enviado, entregado, cancelado

    subtotal NUMERIC(12,2) NOT NULL,
    descuento NUMERIC(12,2) DEFAULT 0,
    total NUMERIC(12,2) NOT NULL,

    metodo_pago VARCHAR(50),
    fecha_pago TIMESTAMP,
    fecha_envio TIMESTAMP,
    fecha_entrega TIMESTAMP,

    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 13. Items de la orden, Esto es el historial de compra real, aquí se congela todo. 
-- Porque si el producto cambia en el futuro, la orden NO debe cambiar.
CREATE TABLE order_items (
    id_orden INTEGER REFERENCES orders(id_orden) ON DELETE CASCADE,
    id_variante INTEGER REFERENCES product_variants(id_variante),
    nombre_producto VARCHAR(150) NOT NULL,
    sku VARCHAR(50),
    cantidad INTEGER NOT NULL,
    precio_unitario NUMERIC(12,2) NOT NULL,
    total NUMERIC(12,2) NOT NULL,
    PRIMARY KEY (id_orden, id_variante)
);

-- 14. Tabla de pagos 
CREATE TABLE pagos (
    id_pago INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_orden INTEGER REFERENCES orders(id_orden) ON DELETE CASCADE,
    proveedor_pago VARCHAR(50), -- Stripe, MercadoPago, PayPal
    referencia_externa VARCHAR(255),
    monto NUMERIC(12,2),
    estado VARCHAR(30), -- aprobado, rechazado, pendiente
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 15. Reseñas para un producto
CREATE TABLE reviews (
    id_review INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_producto INTEGER REFERENCES products(id_producto) ON DELETE CASCADE,
    id_usuario INTEGER REFERENCES users(id_usuario),
    calificacion INTEGER CHECK (calificacion BETWEEN 1 AND 5),
    comentario TEXT,
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 16. Movimientos de inventario, esto para poder llevar un control de lo que son las ventas o movimientos de productos  
CREATE TABLE inventory_movements (
    id_movimiento INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_variante INTEGER REFERENCES product_variants(id_variante),
    tipo VARCHAR(20), -- entrada, salida, ajuste
    cantidad INTEGER NOT NULL,
    referencia VARCHAR(100), -- id_orden, compra proveedor, ajuste manual
    fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Funcion 0
CREATE OR REPLACE FUNCTION get_all_products()
RETURNS TABLE (
    id_producto INTEGER,
    producto VARCHAR,
    descripcion TEXT,
    activo BOOLEAN,
    fecha_creacion TIMESTAMP,
    marca VARCHAR,
	imagen_marca VARCHAR,
    categoria VARCHAR,
    categoria_padre VARCHAR,
    variantes JSONB
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT 
        p.id_producto,
        p.nombre AS producto,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre AS marca,
		m.imagen,
        c.nombre AS categoria,
        cp.nombre AS categoria_padre,

        COALESCE(
            jsonb_agg(
                DISTINCT jsonb_build_object(
                    'id_variante', v.id_variante,
                    'sku', v.sku,
                    'precio', v.precio,
                    'stock', v.stock,
                    'imagenes', v.imagenes,
                    'atributos',
                        (
                            SELECT jsonb_object_agg(a.nombre, vav.valor)
                            FROM variant_attribute_values vav
                            JOIN attributes a 
                                ON a.id_atributo = vav.id_atributo
                            WHERE vav.id_variante = v.id_variante
                        )
                )
            ) FILTER (WHERE v.id_variante IS NOT NULL),
            '[]'::jsonb
        ) AS variantes

    FROM products p
    LEFT JOIN marcas m 
        ON m.id_marca = p.id_marca
    LEFT JOIN categories c 
        ON c.id_categoria = p.id_categoria
    LEFT JOIN categories cp 
        ON cp.id_categoria = c.id_padre
    LEFT JOIN product_variants v 
        ON v.id_producto = p.id_producto

    GROUP BY 
        p.id_producto,
        p.nombre,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre,
		m.imagen,
        c.nombre,
        cp.nombre;

END;
$$;

-- Funcion 1
CREATE OR REPLACE FUNCTION get_inventory_products()
RETURNS TABLE (
    id_producto        INT,
    producto           TEXT,
    activo             BOOLEAN,
    precio             NUMERIC,
    stock              NUMERIC,
    marca              TEXT,
	imagen			   TEXT,
    fecha_creacion     TIMESTAMP
)
LANGUAGE sql
AS $$
    SELECT 
        t.id_producto,
        t.producto,
        t.activo,
        t.precio,
        t.stock,
        t.marca,
		t.imagen,
        t.fecha_creacion
    FROM (
        SELECT DISTINCT ON (p.id_producto)
            p.id_producto,
            p.nombre AS producto,
            p.activo,
            v.precio,
            v.stock,
            m.nombre AS marca,
			m.imagen,
            p.fecha_creacion,

            -- prioridad interna
            CASE
                WHEN p.activo = false THEN 1
                WHEN v.stock < 5 THEN 2
                ELSE 3
            END AS prioridad

        FROM products p
        LEFT JOIN marcas m 
            ON m.id_marca = p.id_marca
        LEFT JOIN product_variants v 
            ON v.id_producto = p.id_producto

        -- elegimos variante con menor stock
        ORDER BY p.id_producto, v.stock ASC
    ) t
    ORDER BY t.prioridad, t.stock ASC;
$$;

CREATE OR REPLACE FUNCTION get_products_with_variants_without_attributes()
RETURNS TABLE (
    id_producto INT,
    nombre TEXT,
    descripcion TEXT,
    activo BOOLEAN,
    fecha_creacion TIMESTAMP
)
LANGUAGE sql
AS $$
    SELECT 
        p.id_producto,
        p.nombre,
        p.descripcion,
        p.activo,
        p.fecha_creacion
    FROM products p
    WHERE EXISTS (
        SELECT 1
        FROM product_variants v
        WHERE v.id_producto = p.id_producto
    )
    AND EXISTS (
        SELECT 1
        FROM product_variants v
        WHERE v.id_producto = p.id_producto
          AND NOT EXISTS (
              SELECT 1
              FROM variant_attribute_values vav
              WHERE vav.id_variante = v.id_variante
          )
    ) order by id_producto;
$$;

-- Funcion 5
-- funcion para actualizar datos de inventario
CREATE OR REPLACE FUNCTION update_product_variant(
    p_id_producto INT,
    p_id_variante INT,
    p_precio NUMERIC,
    p_stock INT,
    p_activo BOOLEAN
)
RETURNS INT
LANGUAGE plpgsql
AS $$
DECLARE
    v_rows_affected INT := 0;
    v_temp INT;
BEGIN

    -- Update variante
    UPDATE product_variants
    SET 
        precio = p_precio,
        stock = p_stock
    WHERE id_variante = p_id_variante
      AND id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    -- Update producto
    UPDATE products
    SET 
        activo = p_activo
    WHERE id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    RETURN v_rows_affected;

END;
$$;

-- Actualiza los datos base de un producto
CREATE OR REPLACE FUNCTION update_full_product(
    p_id_producto INT,
    p_id_marca INT,
    p_id_categoria INT,
    p_nombre TEXT,
    p_descripcion TEXT
)
RETURNS TABLE (
    updated_product INT
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_rows_product INT := 0;
BEGIN

    -- Update producto
    UPDATE products
    SET id_marca = p_id_marca,
        id_categoria = p_id_categoria,
        nombre = p_nombre,
        descripcion = p_descripcion,
        fecha_actualizacion = NOW()
    WHERE id_producto = p_id_producto;

    GET DIAGNOSTICS v_rows_product = ROW_COUNT;


    RETURN QUERY
    SELECT v_rows_product;

END;
$$;

-- Funcion 7
-- Actualiza los datos de una variante
CREATE OR REPLACE FUNCTION update_product_var_attr(
    p_id_producto INT,

    p_id_variante INT,
    p_sku TEXT,
    p_imagenes JSONB,

    p_id_atributo INT,
    p_valor TEXT
)
RETURNS TABLE (
    updated_variant INT,
    updated_attribute INT
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_rows_variant INT := 0;
    v_rows_attribute INT := 0;
BEGIN

    -- Update variante
    UPDATE product_variants
    SET sku = p_sku,
        imagenes = p_imagenes
    WHERE id_variante = p_id_variante
      AND id_producto = p_id_producto;

    GET DIAGNOSTICS v_rows_variant = ROW_COUNT;

    -- Update atributo
    UPDATE variant_attribute_values
    SET valor = p_valor
    WHERE id_variante = p_id_variante
      AND id_atributo = p_id_atributo;

    GET DIAGNOSTICS v_rows_attribute = ROW_COUNT;

    RETURN QUERY
    SELECT v_rows_variant, v_rows_attribute;
END;
$$;

CREATE OR REPLACE FUNCTION core.get_all_products()
RETURNS TABLE (
    id_producto INTEGER,
    producto VARCHAR,
    descripcion TEXT,
    activo BOOLEAN,
    fecha_creacion TIMESTAMP,
    marca VARCHAR,
    imagen_marca VARCHAR,
    categoria VARCHAR,
    categoria_padre VARCHAR,
    variantes JSONB
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT 
        p.id_producto,
        p.nombre AS producto,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre AS marca,
        m.imagen,
        c.nombre AS categoria,
        cp.nombre AS categoria_padre,
        COALESCE(
            jsonb_agg(
                DISTINCT jsonb_build_object(
                    'id_variante', v.id_variante,
                    'sku', v.sku,
                    'precio', v.precio,
                    'stock', v.stock,
                    'imagenes', v.imagenes,
                    'atributos',
                        (
                            SELECT jsonb_object_agg(a.nombre, vav.valor)
                            FROM core.variant_attribute_values vav
                            JOIN core.attributes a 
                                ON a.id_atributo = vav.id_atributo
                            WHERE vav.id_variante = v.id_variante
                        )
                )
            ) FILTER (WHERE v.id_variante IS NOT NULL),
            '[]'::jsonb
        )
    FROM core.products p
    LEFT JOIN core.marcas m 
        ON m.id_marca = p.id_marca
    LEFT JOIN core.categories c 
        ON c.id_categoria = p.id_categoria
    LEFT JOIN core.categories cp 
        ON cp.id_categoria = c.id_padre
    LEFT JOIN core.product_variants v 
        ON v.id_producto = p.id_producto
    GROUP BY 
        p.id_producto,
        p.nombre,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre,
        m.imagen,
        c.nombre,
        cp.nombre;

$$;

    FROM core.products p
    WHERE NOT EXISTS (

        FROM core.product_variants v
        WHERE v.id_producto = p.id_producto
    );
$$;

-- ===============================
CREATE OR REPLACE FUNCTION core.update_product_variant(
    p_id_producto INT,
    p_id_variante INT,
    p_precio NUMERIC,
    p_stock INT,
    p_activo BOOLEAN
)
RETURNS INT
LANGUAGE plpgsql
AS $$
DECLARE
    v_rows_affected INT := 0;
    v_temp INT;
BEGIN
    UPDATE core.product_variants
    SET precio = p_precio,
        stock = p_stock
    WHERE id_variante = p_id_variante
      AND id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    UPDATE core.products
    SET activo = p_activo
    WHERE id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    RETURN v_rows_affected;
END;
$$;

CREATE OR REPLACE FUNCTION get_all_products()
RETURNS TABLE (
    id_producto INTEGER,
    producto VARCHAR,
    descripcion TEXT,
    activo BOOLEAN,
    fecha_creacion TIMESTAMP,
    marca VARCHAR,
	imagen_marca VARCHAR,
    categoria VARCHAR,
    categoria_padre VARCHAR,
    variantes JSONB
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT 
        p.id_producto,
        p.nombre AS producto,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre AS marca,
		m.imagen,
        c.nombre AS categoria,
        cp.nombre AS categoria_padre,

        COALESCE(
            jsonb_agg(
                DISTINCT jsonb_build_object(
                    'id_variante', v.id_variante,
                    'sku', v.sku,
                    'precio', v.precio,
                    'stock', v.stock,
                    'imagenes', v.imagenes,
                    'atributos',
                        (
                            SELECT jsonb_object_agg(a.nombre, vav.valor)
                            FROM variant_attribute_values vav
                            JOIN attributes a 
                                ON a.id_atributo = vav.id_atributo
                            WHERE vav.id_variante = v.id_variante
                        )
                )
            ) FILTER (WHERE v.id_variante IS NOT NULL),
            '[]'::jsonb
        ) AS variantes

    FROM products p
    LEFT JOIN marcas m 
        ON m.id_marca = p.id_marca
    LEFT JOIN categories c 
        ON c.id_categoria = p.id_categoria
    LEFT JOIN categories cp 
        ON cp.id_categoria = c.id_padre
    LEFT JOIN product_variants v 
        ON v.id_producto = p.id_producto

    GROUP BY 
        p.id_producto,
        p.nombre,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre,
		m.imagen,
        c.nombre,
        cp.nombre;

END;
$$;

CREATE OR REPLACE FUNCTION get_inventory_products()
RETURNS TABLE (
    id_producto        INT,
    producto           TEXT,
    activo             BOOLEAN,
    precio             NUMERIC,
    stock              NUMERIC,
    marca              TEXT,
	imagen			   TEXT,
    fecha_creacion     TIMESTAMP
)
LANGUAGE sql
AS $$
    SELECT 
        t.id_producto,
        t.producto,
        t.activo,
        t.precio,
        t.stock,
        t.marca,
		t.imagen,
        t.fecha_creacion
    FROM (
        SELECT DISTINCT ON (p.id_producto)
            p.id_producto,
            p.nombre AS producto,
            p.activo,
            v.precio,
            v.stock,
            m.nombre AS marca,
			m.imagen,
            p.fecha_creacion,

            -- prioridad interna
            CASE
                WHEN p.activo = false THEN 1
                WHEN v.stock < 5 THEN 2
                ELSE 3
            END AS prioridad

        FROM products p
        LEFT JOIN marcas m 
            ON m.id_marca = p.id_marca
        LEFT JOIN product_variants v 
            ON v.id_producto = p.id_producto

        -- elegimos variante con menor stock
        ORDER BY p.id_producto, v.stock ASC
    ) t
    ORDER BY t.prioridad, t.stock ASC;
$$;

-- Funcion 5
-- funcion para actualizar datos de inventario
CREATE OR REPLACE FUNCTION update_product_variant(
    p_id_producto INT,
    p_id_variante INT,
    p_precio NUMERIC,
    p_stock INT,
    p_activo BOOLEAN
)
RETURNS INT
LANGUAGE plpgsql
AS $$
DECLARE
    v_rows_affected INT := 0;
    v_temp INT;
BEGIN

    -- Update variante
    UPDATE product_variants
    SET 
        precio = p_precio,
        stock = p_stock
    WHERE id_variante = p_id_variante
      AND id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    -- Update producto
    UPDATE products
    SET 
        activo = p_activo
    WHERE id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    RETURN v_rows_affected;

END;
$$;

    v_rows_variant INT := 0;
    v_rows_attribute INT := 0;
BEGIN

    -- Update variante
    UPDATE product_variants
    SET sku = p_sku,
        imagenes = p_imagenes
    WHERE id_variante = p_id_variante
      AND id_producto = p_id_producto;

    -- Update atributo
    UPDATE variant_attribute_values
    SET valor = p_valor
    WHERE id_variante = p_id_variante

    GET DIAGNOSTICS v_rows_attribute = ROW_COUNT;

    RETURN QUERY
    SELECT v_rows_variant, v_rows_attribute;
END;
$$;

CREATE OR REPLACE FUNCTION core.get_all_products()
RETURNS TABLE (
    id_producto INTEGER,
    producto VARCHAR,
    descripcion TEXT,
    activo BOOLEAN,
    fecha_creacion TIMESTAMP,
    marca VARCHAR,
    imagen_marca VARCHAR,
    categoria VARCHAR,
    categoria_padre VARCHAR,
    variantes JSONB
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT 
        p.id_producto,
        p.nombre AS producto,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre AS marca,
        m.imagen,
        c.nombre AS categoria,
        cp.nombre AS categoria_padre,
        COALESCE(
            jsonb_agg(
                DISTINCT jsonb_build_object(
                    'id_variante', v.id_variante,
                    'sku', v.sku,
                    'precio', v.precio,
                    'stock', vi.stock_ac,
                    'imagenes', v.imagenes,
                    'atributos',
                        (
                            SELECT jsonb_object_agg(a.nombre, vav.valor)
                            FROM core.variant_attribute_values vav
                            JOIN core.attributes a 
                                ON a.id_atributo = vav.id_atributo
                            WHERE vav.id_variante = v.id_variante
                        )
                )
            ) FILTER (WHERE v.id_variante IS NOT NULL),
            '[]'::jsonb
        )
    FROM core.products p
    LEFT JOIN core.marcas m 
        ON m.id_marca = p.id_marca
    LEFT JOIN core.categories c 
        ON c.id_categoria = p.id_categoria
    LEFT JOIN core.categories cp 
        ON cp.id_categoria = c.id_padre
    LEFT JOIN core.product_variants v 
        ON v.id_producto = p.id_producto
    GROUP BY 
        p.id_producto,
        p.nombre,
        p.descripcion,
        p.activo,
        p.fecha_creacion,
        m.nombre,
        m.imagen,
        c.nombre,
        cp.nombre;
END;
$$;

CREATE OR REPLACE FUNCTION core.get_inventory_products()
RETURNS TABLE (
    id_producto INT,
    producto TEXT,
    activo BOOLEAN,
    precio NUMERIC,
    stock NUMERIC,
    marca TEXT,
    imagen TEXT,
    fecha_creacion TIMESTAMP
)
LANGUAGE sql
AS $$
    SELECT 
        t.id_producto,
        t.producto,
        t.activo,
        t.precio,
        t.stock,
        t.marca,
        t.imagen,
        t.fecha_creacion
    FROM (
        SELECT DISTINCT ON (p.id_producto)
            p.id_producto,
            p.nombre AS producto,
            p.activo,
            v.precio,
            v.stock,
            m.nombre AS marca,
            m.imagen,
            p.fecha_creacion,
            CASE
                WHEN p.activo = false THEN 1
                WHEN v.stock < 5 THEN 2
                ELSE 3
            END AS prioridad
        FROM core.products p
        LEFT JOIN core.marcas m 
            ON m.id_marca = p.id_marca
        LEFT JOIN core.product_variants v 
            ON v.id_producto = p.id_producto
        ORDER BY p.id_producto, v.stock ASC
    ) t
    ORDER BY t.prioridad, t.stock ASC;
$$;

RETURNS TABLE (
    id_producto INT,
    nombre TEXT,
    descripcion TEXT,
    activo BOOLEAN,
    fecha_creacion TIMESTAMP
)

AS $$

          )
    )

$$;

-- ===============================
CREATE OR REPLACE FUNCTION core.update_product_variant(
    p_id_producto INT,
    p_id_variante INT,
    p_precio NUMERIC,
    p_stock INT,
    p_activo BOOLEAN
)
RETURNS INT
LANGUAGE plpgsql
AS $$
DECLARE
    v_rows_affected INT := 0;
    v_temp INT;
BEGIN
    UPDATE core.product_variants
    SET precio = p_precio,
        stock = p_stock
    WHERE id_variante = p_id_variante
      AND id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    UPDATE core.products
    SET activo = p_activo
    WHERE id_producto = p_id_producto;

    GET DIAGNOSTICS v_temp = ROW_COUNT;
    v_rows_affected := v_rows_affected + v_temp;

    RETURN v_rows_affected;
END;
$$;
